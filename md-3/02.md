## 模块化

```js
// util1.js

export default {
    a: 100
}


// util2.js

export function fn1 {
    console.log('fn1');
}
export function fn2 {
    console.log('fn2');
}


// index.js
import util1 from './util1.js';
import { fn1, fn2 } from './util1.js';

console.log(util1);

fn(1);
fn(2);
```
